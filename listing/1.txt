package main
 
import (
    "fmt"
)
 
func main() {
    a := [5]int{76, 77, 78, 79, 80}
    var b []int = a[1:4]
    fmt.Println(b)
}

Программа выведет [77 78 79] . Переменной b присваивается новый срез, который имеет len = 3 и cap = 4 и содержит элементы массива a с индексами 1,2,3.
Но срез b также ссылается на элементы массива a и при изменении элементов среза b будут также меняться элементы массива a
